plt.show()

# group the patients in 10 and average their inflammation levels
# 0..9 : group0, 10..19: group1 ... 50..59: group5

dataBy10=np.empty([6, 40])
for i in range(6):
    dataBy10[i] = np.mean(data[10*i + 0 : 10*i + 10, :], axis=0)
    
for i in range(6):
    plt.plot(dataBy10[i], label='obs. og patient group{}'.format(i))

plt.xlabel('The days of observation')
plt.ylabel('Mean inflammation level')
plt.legend(bbox_to_anchor=(1.6, 1.01))
plt.show()
