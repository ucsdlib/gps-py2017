# functions (3)
import numpy as np

def normalize(lst):
    zlst = np.empty(len(lst))
    mean = np.mean(lst)
    std = np.std(lst)
    for idx, x in enumerate(lst):
        zlst[idx] = (x - mean)/std
    return zlst

print(normalize(np.array([1,3,5,7,9])))

import numpy as np
import matplotlib.pyplot as plt

# Create a function that shows a stacked bar plot of
# gun homicides and total homicides

def plot_stacked(df, limit=10):
    df = df.ix[:,['Homicides', 'Gun Homicides']][:limit]
    df['Total Homicides'] = df['Homicides'] - df['Gun Homicides']
    del df['Homicides']
    df.plot(kind='bar', stacked=True, sort_columns=True)
    plt.ylabel("per 100k")
    plt.show()
